def split_text_by_sections(text: str, start_headers: List[str], end_headers: List[str]) -> Tuple[str, str, str]:
    """Generic function to split text by section headers"""
    start_idx = -1
    for header in start_headers:
        match = re.search(header, text)
        if match:
            start_idx = match.start()
            break
    
    if start_idx == -1:
        return text, "", ""
    
    end_idx = len(text)
    for header in end_headers:
        match = re.search(header, text[start_idx:])
        if match:
            end_idx = start_idx + match.start()
            break
    
    return text[:start_idx].strip(), text[start_idx:end_idx].strip(), text[end_idx:].strip()
